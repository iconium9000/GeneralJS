<str> = '<txt>' | "<txt>" | `<txt>`
<com> = //<txt>\n | /*<txt>*/
<empty> = <com>
<tok> = [native]
<wrd> = <tok><txt><tok>
  if (isNaN(parseFloat(<wrd>))) <id> = <wrd>
  else <num> = parseFloat(<wrd>)

<stat1> = <tok> <stat1> | <num> <stat1> | <id> <stat1> | <sub> <stat1> | <empty>
  <par> = (<stat1>)
  <brk> = [<stat1>]
  <brc> = {<stat1>}
  <sub> = <par> | <brk> | <brc>

<val> = <num> | <id> | <sub>
<fun> = <val1> <val1> | <val1> <fun>
<val2> = <val1> | <fun>
<val3> = <tok prefx> <val2> | <val2> <tok pstfx>
<val4> = <val2> | <val3>
<val5> = <val5> <tok op1> <val4>
